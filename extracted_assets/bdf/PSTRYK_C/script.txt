void Group.HotSpot_Holder.MouseMoveIn(int rectID)
{
   switch( rectID )
   {
      case 0:   // wysuniecie poprawnej strony
         SetCursor(0);
         SetTimer(0);
         Group.Sound_Holder.Stop(0);

         Sprite_Holder.GotoXY( 0, 40, 0 );
         Group.Sound_Holder.Stop(9);
         Group.Sound_Holder.PlayDSound(8);

         break;
      case 11:   //wysun toolbar
         Group.Sprite_Holder.DisableTimers();

         Group.Sprite_Holder.GotoXY(15,480,0);
         Group.Sprite_Holder.GotoXY(16,560,329);
         Group.Sprite_Holder.GotoXY(17,557,404);
         Group.Sprite_Holder.GotoXY(18,544,14);
         Group.Sprite_Holder.GotoXY(19,584,14);

         Group.Sprite_Holder.SynchronizeTimers();
         Group.Sprite_Holder.EnableTimers();
         Group.Sound_Holder.PlayDSound(5);
         SetCursor(0);
         break;
      case 12:
         SetCursor(1);
         if(toolbar)
         {
            Group.Sprite_Holder.DisableTimers();

            Group.Sprite_Holder.GotoXY(15,640,0);
            Group.Sprite_Holder.GotoXY(16,721,329);
            Group.Sprite_Holder.GotoXY(17,717,404);
            Group.Sprite_Holder.GotoXY(18,704,14);
            Group.Sprite_Holder.GotoXY(19,744,14);

            Group.Sprite_Holder.SynchronizeTimers();
            Group.Sprite_Holder.EnableTimers();
            Group.Sound_Holder.PlayDSound(5);
            toolbar=0;
         }
         Group.HotSpot_Holder.DisableHotSpot(12);
         break;
   }
}

}

void Sprite_Holder.MouseClickOnDown(int spriteID)
{
   if( (spriteID >= 1) && (spriteID <= 10) )
   {
      int sprite = spriteID-1;
      if( roznice[sprite] == 0 )
      {
         roznice[sprite]=1;
         Group.Sound_Holder.PlayDSound(1);
         Sprite_Holder.ShowSprite(spriteID+10);
         Sprite_Holder.HideSprite(spriteID);

         //Sprite_Holder.ChangePhase( spriteID, 1 );

         Group.Sprite_Holder.ChangePhase( poprawnie, 1 );
         poprawnie++;

         if( poprawnie >= 10 )
         {
            Group.Sound_Holder.PlayDSound(3);
         }
      }
      
   }
}

void Bitmap_Holder.MouseClickOnDown(int bitmapID)
{
   switch( bitmapID )
   {
      case 0: // blad
         if( poprawnie >= 10 ) return;

         Group.Sound_Holder.PlayDSound(2);
         Group.Sprite_Holder.ChangePhase( blad+10, 1 );
         blad++;

         if( blad >= 5 )
         {
            // Popelniono 5 bledow - reset
            Group.Sound_Holder.PlayDSound(4);

            blad = 0;
            poprawnie = 0;

            int index = 0;
            while( index < 10 )
            {
               roznice[index]=0;
               Group.Sprite_Holder.ChangePhase( index, 0 );
               if( index < 5 )
                  Group.Sprite_Holder.ChangePhase( index+10, 0 );
               //Sprite_Holder.ChangePhase( index+1, 0 );
               Sprite_Holder.HideSprite( index+11 );
               Sprite_Holder.ShowSprite( index+1 );
               index++;
            }
         }
   }
}

void Group.Sprite_Holder.MouseClickOnDown(int spriteID)
{
   if(spriteID>15)
   {
      Group.Sprite_Holder.ChangePhase(spriteID,1);
      switch(spriteID)
      {
         case 16:
            Group.Sound_Holder.PlayDSound(6);
            break;
         case 17:
            Group.Sound_Holder.PlayDSound(7);
            break;
         case 18:   //parent window
            Group.Sound_Holder.PlayDSound(11);
            break;
         case 19:   //plansza glowna
            Group.Sound_Holder.PlayDSound(12);
            break;
      }
      Group.Sprite_Holder.DisableTimers();

      Group.Sprite_Holder.GotoXY(15,640,0);
      Group.Sprite_Holder.GotoXY(16,721,329);
      Group.Sprite_Holder.GotoXY(17,717,404);
      Group.Sprite_Holder.GotoXY(18,704,14);
      Group.Sprite_Holder.GotoXY(19,744,14);

      Group.Sprite_Holder.SynchronizeTimers();
      Group.Sprite_Holder.EnableTimers();
      Group.Sound_Holder.PlayDSound(5);
      toolbar=0;
      Group.HotSpot_Holder.DisableHotSpot(11);
      toolbarCase=spriteID;
   }
}


void Group.Sprite_Holder.InPlace(int spriteID,int x,int y)
{
   if(spriteID==15)
   {
      if(x==480)
      {
         toolbar=1;
         Group.HotSpot_Holder.EnableHotSpot(12);
         SetCursor(0);
      }
      else
      {
         Group.Sprite_Holder.ChangePhase(16,0);
         Group.Sprite_Holder.ChangePhase(17,0);
         Group.Sprite_Holder.ChangePhase(18,0);
         Group.Sprite_Holder.ChangePhase(19,0);
         switch(toolbarCase)
         {
            case 18:	//lewo
               LoadPage("malus_c.bdf");
               break;
            case 19:	//prawo
               LoadPage("plotki_c.bdf");
               break;
            case 16:   //do planszy glownej   
               LoadPage( "pstryk.bdf" );
               break;
            case 17:   //Plansza wyboru
               LoadPage("Wyborw.bdf");
               break;
         }
      }
   }
}
